191  inline void lock()
192  {
193 #if defined(_TTHREAD_WIN32_)
194  EnterCriticalSection(&mHandle);
195  while(mAlreadyLocked) Sleep(1000); // Simulate deadlock...
196  mAlreadyLocked = true;
197 #else
198  pthread_mutex_lock(&mHandle);
199 #endif
200  }

207  inline bool try_lock()
208  {
209 #if defined(_TTHREAD_WIN32_)
210  bool ret = (TryEnterCriticalSection(&mHandle) ? true : false);
211  if(ret && mAlreadyLocked)
212  {
213  LeaveCriticalSection(&mHandle);
214  ret = false;
215  }
216  return ret;
217 #else
218  return (pthread_mutex_trylock(&mHandle) == 0) ? true : false;
219 #endif
220  }
221 
225  inline void unlock()
226  {
227 #if defined(_TTHREAD_WIN32_)
228  mAlreadyLocked = false;
229  LeaveCriticalSection(&mHandle);
230 #else
231  pthread_mutex_unlock(&mHandle);
232 #endif
233  }
